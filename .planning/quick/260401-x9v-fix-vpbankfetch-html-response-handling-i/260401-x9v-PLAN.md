```javascript
async function fetchJson(url) {
  const response = await vpbankFetch(url);
  const text = await response.text();
  const trimmed = text.trim();
  if (trimmed.startsWith('<')) {
    const err = new Error(`External API returned HTML instead of JSON (status ${response.status}).`);
    err.status = response.status;
    err.url = url;
    throw err;
  }
  let data;
  try {
    data = JSON.parse(trimmed);
  } catch (e) {
    const err = new Error(`External API response is not valid JSON (status ${response.status}).`);
    err.status = response.status;
    err.url = url;
    throw err;
  }
  return { response, data };
}
```
Return type: `{ response: Response, data: any }`
